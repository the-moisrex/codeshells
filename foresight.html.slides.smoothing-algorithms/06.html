class ISmoothingAlgorithm {
  public:
    virtual void initialize() = 0;
    virtual context_action process(Context& ctx) = 0;
    virtual void reset() = 0;
    virtual void setParameters(const Parameters& params) = 0;
    virtual ~ISmoothingAlgorithm() = default;
};
